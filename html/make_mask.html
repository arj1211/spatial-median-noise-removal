Acquire the pixels under mask based on image, center_pixel position, and mask size.
Author: Arjun Bawa, Last modified on Dec 4, 2021
function [mask, bounds] = make_mask(img, pos, msize)
start_h = max([pos(1) - floor(msize/2), 1]);
end_h = min([pos(1) + floor(msize/2), size(img, 1)]);
start_w = max([pos(2) - floor(msize/2), 1]);
end_w = min([pos(2) + floor(msize/2), size(img, 2)]);
mask = permute(img(start_h:end_h, start_w:end_w, :), [3,1,2]);
mask = transpose(double([mask(1, :); mask(2, :); mask(3, :)]));
bounds = [start_h, end_h; start_w,end_w];
end
